Пример работы

import asyncio
import vkbee

async def main(loop):
    token = "сюдатокен"
    vk_s = vkbee.VkApi(token, loop=loop)
    vk = vk_s.get_api()
    await vk_s.messages.send(
        chat_id=1,
        message='VKBEE',
        random_id=0
    )
loop = asyncio.get_event_loop()
loop.run_until_complete(main(loop))
loop.close()
